🤔 Why combine LangGraph and CrewAI

LangGraph and CrewAI don't compete — they complement each other. LangGraph excels atmacro flow control(when to do what, under what conditions, with checkpoints). CrewAI excels atcollaborative execution(teams of agents doing in-depth analysis). Together they cover the full spectrum.

🎯Separation of responsibilities

🕸️ LangGraph takes care of:

  • •Order and conditions for executing the steps
  • •Checkpoints before and after each phase
  • •Retry logic in case of failure
  • •Human-in-the-loop when needed
  • •Progress streaming to the user

👥 CrewAI takes care of:

  • •In-depth analysis by a team of experts
  • •Collaboration between agents with clear roles
  • •Structured output in Pydantic
  • •Automatic delegation of sub-tasks
  • •Generation of rich and structured reports

💡The principle of intelligent combination

Think of LangGraph as theorchestra directorand CrewAI as agroup of expert musicians. The director decides when each group plays and maintains the overall rhythm. The musicians perform the score masterfully within their domain. Neither replaces the other.

👥 CrewAI as a LangGraph node

The implementation pattern: a LangGraph node function that instantiates a Crew, callscrew.kickoff()with inputs from the state, and returns the result back to the state.

🔨Crew node implementation

from langgraph.graph import StateGraph, START, END
from crewai import Agent, Task, Crew, Process
from typing import TypedDict

# LangGraph State
class DueDiligenceState(TypedDict):
    empresa: str
    setor: str
    dados_coletados: dict
    analise_crew: dict  # Crew output goes here
    validacao: dict
    relatorio_final: str
    aprovado: bool

# ===== CREW FACTORY =====
def criar_crew_analise(empresa: str, dados: dict) -> Crew:
    """Cria e retorna uma Crew configurada para a análise."""
    analista_financeiro = Agent(
        role="Financial Analyst",
        goal=f"Analise a saúde financeira de {empresa}",
        backstory="Especialista em valuation e análise fundamentalista com 15 anos.",
        llm="gpt-4o"
    )
    analista_risco = Agent(
        role="Risk Analyst",
        goal=f"Identifique riscos críticos para {empresa}",
        backstory="Ex-consultor de risco com foco em M&A e due diligence.",
        llm="gpt-4o"
    )
    analista_mercado = Agent(
        role="Market Analyst",
        goal=f"Avalie o posicionamento de mercado de {empresa}",
        backstory="Especialista em análise competitiva e estratégia de mercado.",
        llm="gpt-4o-mini"
    )
    task_financeira = Task(
        description=f"""Analise os dados financeiros de {empresa}:
        Dados: {dados}
        Foco: rentabilidade, liquidez, endividamento""",
        expected_output="Análise financeira com score 1-10 e principais alertas",
        agent=analista_financeiro
    )
    task_risco = Task(
        description=f"Identifique top 5 riscos para investimento em {empresa}",
        expected_output="Lista de riscos com probabilidade e impacto",
        agent=analista_risco,
        context=[task_financeira]
    )
    task_mercado = Task(
        description=f"Avalie posição competitiva de {empresa} no mercado",
        expected_output="Análise de posicionamento com oportunidades e ameaças",
        agent=analista_mercado
    )
    return Crew(
        agents=[analista_financeiro, analista_risco, analista_mercado],
        tasks=[task_financeira, task_risco, task_mercado],
        process=Process.sequential,
        verbose=False
    )

# ===== LANGGRAPH NODE RUNNING THE CREW =====
def executar_analise_crew(state: DueDiligenceState) -> dict:
    """Nó do LangGraph que instancia e executa a Crew."""
    empresa = state["empresa"]
    dados = state["dados_coletados"]

    # Create the Crew with the state context
    crew = criar_crew_analise(empresa, dados)

    # Execute with timeout (important!)
    try:
        resultado = crew.kickoff(inputs={
            "empresa": empresa,
            "setor": state["setor"]
        })
        return {
            "analise_crew": {
                "resultado": resultado.raw,
                "aprovado": True,
                "custo_tokens": str(resultado.token_usage)
            }
        }
    except Exception as e:
        return {
            "analise_crew": {
                "resultado": f"Erro na análise: {str(e)}",
                "aprovado": False
            }
        }

📦 Passing state between frameworks

A border between LangGraph and CrewAIThis is where bugs happen. The integration node needs to convert the LangGraph state to the Crew input format, and serialize the Crew output back to the state — with validation on both sides.

🔀Clear interface agreement

from pydantic import BaseModel

# INPUT contract for the Crew
class CrewInput(BaseModel):
    empresa: str
    setor: str
    dados_financeiros: dict
    periodo_analise: str = "últimos 3 anos"

# Crew OUTPUT contract
class CrewOutput(BaseModel):
    score_financeiro: float        # 0.0 to 10.0
    principais_riscos: list[str]
    oportunidades: list[str]
    recomendacao: str              # "INVEST" | "WAIT" | "REFUSE"
    confianca: float               # 0.0 to 1.0

# Node with explicit conversion and validation
def executar_analise_crew(state: DueDiligenceState) -> dict:
    # 1. Convert state → CrewInput with validation
    try:
        crew_input = CrewInput(
            empresa=state["empresa"],
            setor=state["setor"],
            dados_financeiros=state["dados_coletados"].get("financeiros", {}),
        )
    except Exception as e:
        return {"analise_crew": {"erro": f"Input inválido: {e}", "aprovado": False}}

    # 2. Run the Crew
    crew = criar_crew_analise(crew_input.empresa, crew_input.model_dump())
    resultado = crew.kickoff()

    # 3. Validate and convert output → state
    try:
        # Final task with output_pydantic=CrewOutput
        analise = resultado.pydantic  # Typed CrewOutput
        return {
            "analise_crew": {
                "score": analise.score_financeiro,
                "riscos": analise.principais_riscos,
                "recomendacao": analise.recomendacao,
                "aprovado": analise.recomendacao != "RECUSAR"
            }
        }
    except Exception as e:
        return {"analise_crew": {"raw": resultado.raw, "erro": str(e)}}

⚠️Common Integration Pitfalls

  • ✗Pass non-serializable objects between frameworks (use primitive types)
  • ✗Not validating the Crew output before placing it in state
  • ✗Create the Crew within the graph without error handling
  • ✗No timeout: Crew can take indefinitely
  • ✓Use try/except throughout the integration node
  • ✓Pydantic on both sides: Crew input and output

🔷 Reusable integration patterns

Three proven patterns for combining LangGraph and CrewAI — each solves a different category of problem.

1

Crew-on-Node Pattern (Sequential)

A Crew on a single sequential node. The simplest and most common.

grafo.add_node("coleta", coletar_dados)
grafo.add_node("analise", executar_crew)  # Crew here
grafo.add_node("relatorio", gerar_relatorio)
grafo.add_edge(START, "coleta")
grafo.add_edge("coleta", "analise")
grafo.add_edge("analise", "relatorio")
grafo.add_edge("relatorio", END)
2

Parallel Crews Pattern (Map-Reduce)

Multiple Crews running in parallel, results aggregated by an end node.

from langgraph.constants import Send

# Distribute to multiple Crews in parallel
def distribuir_analises(state) -> list:
    return [
        Send("analise_crew", {"empresa": e, **state})
        for e in state["empresas_para_analisar"]
    ]

grafo.add_conditional_edges("preparar", distribuir_analises)
grafo.add_node("analise_crew", executar_crew)  # each company
grafo.add_node("agregar", combinar_resultados) # unites everything
3

Loop Pattern with Crew (Smart Retry)

The loop checks quality and calls the Crew again if necessary.

def checar_qualidade(state) -> str:
    score = state["analise_crew"].get("score", 0)
    tentativas = state.get("tentativas", 0)
    if score >= 7.0 or tentativas >= 3:
        return "relatorio"
    return "analise_crew"  # return to the Crew

grafo.add_conditional_edges("avaliar", checar_qualidade)

🔍 Project: automated due diligence system

The complete project: due diligence system where LangGraph orchestrates the steps (collection → analysis → validation → reporting) and a CrewAI of 3 analysts performs the in-depth analysis step.

🚀Complete due diligence graph

from langgraph.checkpoint.memory import MemorySaver

# LangGraph nodes (without Crew)
def coletar_dados(state: DueDiligenceState) -> dict:
    """Coleta dados públicos da empresa via APIs."""
    empresa = state["empresa"]
    return {
        "dados_coletados": {
            "financeiros": buscar_dados_financeiros(empresa),
            "mercado": buscar_dados_mercado(empresa, state["setor"]),
            "noticias": buscar_noticias_recentes(empresa),
            "juridico": buscar_processos_juridicos(empresa)
        }
    }

def validar_analise(state: DueDiligenceState) -> dict:
    """Valida se a análise da Crew atende critérios mínimos."""
    analise = state["analise_crew"]
    issues = []
    if analise.get("score", 0) == 0:
        issues.append("Score não calculado")
    if not analise.get("riscos"):
        issues.append("Riscos não identificados")
    return {
        "validacao": {
            "aprovada": len(issues) == 0,
            "issues": issues
        }
    }

def gerar_relatorio_final(state: DueDiligenceState) -> dict:
    """Gera relatório executivo consolidado."""
    analise = state["analise_crew"]
    return {
        "relatorio_final": f"""
# Due Diligence: {state['empresa']}

## Executive Summary
**Recomendação:** {analise.get('recomendacao', 'N/A')}
**Score Financeiro:** {analise.get('score', 'N/A')}/10

## Principais Riscos
{chr(10).join(f'- {r}' for r in analise.get('riscos', []))}

## Próximos Passos
{analise.get('proximos_passos', 'Consultar equipe jurídica')}
        """,
        "aprovado": analise.get("recomendacao") == "INVESTIR"
    }

# Routing logic
def rotear_apos_validacao(state: DueDiligenceState) -> str:
    if state["validacao"]["aprovada"]:
        return "relatorio"
    tentativas = state.get("tentativas", 0)
    if tentativas >= 2:
        return "relatorio"  # force exit
    return "analise"        # repeat the Crew

# Assemble the complete graph
grafo = StateGraph(DueDiligenceState)
grafo.add_node("coleta", coletar_dados)
grafo.add_node("analise", executar_analise_crew)  # Crew here
grafo.add_node("validacao", validar_analise)
grafo.add_node("relatorio", gerar_relatorio_final)

grafo.add_edge(START, "coleta")
grafo.add_edge("coleta", "analise")
grafo.add_edge("analise", "validacao")
grafo.add_conditional_edges("validacao", rotear_apos_validacao, {
    "relatorio": "relatorio",
    "analise": "analise"
})
grafo.add_edge("relatorio", END)

# Compile with checkpoint
checkpointer = MemorySaver()
app = grafo.compile(checkpointer=checkpointer)

# Execute
config = {"configurable": {"thread_id": "dd_empresa_xyz_001"}}
resultado = app.invoke({
    "empresa": "TechStartup XYZ",
    "setor": "SaaS B2B",
    "dados_coletados": {},
    "analise_crew": {},
    "validacao": {},
    "relatorio_final": "",
    "aprovado": False
}, config=config)

print(resultado["relatorio_final"])
